def initialize(message, meta=nil)
  @meta = {}

  if message.kind_of?(Hash)
    @meta = message
    message = message[:msg] || message[:message] || message[:user]
  else
    @meta = meta || {}
  end

  super(message)
  @message = message

end

def as_json
  res = {
    :error => {
      :code => self.class.name.split(":").last
    }
  }
  if @meta[:user]
    res[:error][:message] = @meta[:user]
  end
  res
end
