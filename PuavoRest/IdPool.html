def self.id_range(namespace, count)
  pool = get_redis_id_pool()

  if pool.get(namespace).nil?
    pool.set(namespace, 5000)
  end

  _id_range = (1..count).map do
    pool.incr(namespace)
  end

  return _id_range
end
