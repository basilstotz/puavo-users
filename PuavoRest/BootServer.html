def self.current_dn
  return if not running_on?

  if ENV["RACK_ENV"] == "test"
    if !PuavoRest.test_boot_server_dn.nil?
      return PuavoRest.test_boot_server_dn
    end
  end

  PUAVO_ETC.ldap_dn
end
