put "/v3/:organisation/ltsp_servers/:domain" do
  if params["cpu_count"]
    load_avg = params["load_avg"].to_f / params["cpu_count"].to_i
  else
    load_avg = params["load_avg"].to_f
  end

  @m.set(params["domain"], load_avg)

  json "ok" => true
end
